    18тЖТ    18тЖТ/**
    19тЖТ    19тЖТ * Extract the best barcode-like number from OCR text.
    20тЖТ    20тЖТ * Strips non-numeric characters, then picks the longest sequence
    21тЖТ    21тЖТ * that matches common barcode lengths (8, 12, 13, 14 digits).
    22тЖТ    22тЖТ */
    23тЖТ    23тЖТfunction extractBarcodeFromOcr(text: string): string | null {
    24тЖТ    24тЖТ  // Remove obvious noise characters but keep digits and spaces/hyphens
    25тЖТ    25тЖТ  const cleaned = text.replace(/[^\d\s\-]/g, ' ');
    26тЖТ    26тЖТ
    27тЖТ    27тЖТ  // Split into candidate number groups
    28тЖТ    28тЖТ  const candidates = cleaned.split(/\s+/).filter(Boolean);
    29тЖТ    29тЖТ
    30тЖТ    30тЖТ  // Also try the entire string with only digits
    31тЖТ    31тЖТ  const allDigits = cleaned.replace(/\D/g, '');
    32тЖТ    32тЖТ
    33тЖТ    33тЖТ  // Collect all digit-only sequences (also combine adjacent groups)
    34тЖТ    34тЖТ  const digitSequences: string[] = [];
    35тЖТ    35тЖТ  for (const c of candidates) {
    36тЖТ    36тЖТ    const d = c.replace(/\D/g, '');
    37тЖТ    37тЖТ    if (d.length >= 6) digitSequences.push(d);
    38тЖТ    38тЖТ  }
    39тЖТ    39тЖТ  if (allDigits.length >= 6) digitSequences.push(allDigits);
    40тЖТ    40тЖТ
    41тЖТ    41тЖТ  // Deduplicate
    42тЖТ    42тЖТ  const unique = [...new Set(digitSequences)];
    43тЖТ    43тЖТ
    44тЖТ    44тЖТ  // Preferred barcode lengths in order of priority
    45тЖТ    45тЖТ  const preferredLengths = [13, 12, 8, 14];
    46тЖТ    46тЖТ
    47тЖТ    47тЖТ  for (const len of preferredLengths) {
    48тЖТ    48тЖТ    // First: exact length match
    49тЖТ    49тЖТ    const exact = unique.find(s => s.length === len);
    50тЖТ    50тЖТ    if (exact) return exact;
    51тЖТ    51тЖТ
    52тЖТ    52тЖТ    // Second: string that starts or ends with a barcode-length subsequence
    53тЖТ    53тЖТ    for (const seq of unique) {
    54тЖТ    54тЖТ      if (seq.length > len) {
    55тЖТ    55тЖТ        const prefix = seq.substring(0, len);
    56тЖТ    56тЖТ        const suffix = seq.substring(seq.length - len);
    57тЖТ    57тЖТ        // Return prefix if it looks like a valid barcode (starts with non-zero)
    58тЖТ    58тЖТ        if (prefix[0] !== '0' || len === 8) return prefix;
    59тЖТ    59тЖТ        if (suffix[0] !== '0' || len === 8) return suffix;
    60тЖТ    60тЖТ      }
    61тЖТ    61тЖТ    }
    62тЖТ    62тЖТ  }
    63тЖТ    63тЖТ
    64тЖТ    64тЖТ  // Fallback: longest digit sequence >= 7 digits
    65тЖТ    65тЖТ  const sorted = unique.sort((a, b) => b.length - a.length);
    66тЖТ    66тЖТ  const longest = sorted.find(s => s.length >= 7);
    67тЖТ    67тЖТ  if (longest) return longest.substring(0, 14); // cap at 14
    68тЖТ    68тЖТ
    69тЖТ    69тЖТ  return null;
    70тЖТ    70тЖТ}

    73тЖТ    73тЖТ * Pre-process a canvas region for better OCR accuracy:
    74тЖТ    74тЖТ * - Convert to greyscale

    78тЖТ    78тЖТfunction preprocessForOcr(sourceCanvas: HTMLCanvasElement): HTMLCanvasElement {
    79тЖТ    79тЖТ  const w = sourceCanvas.width;
    80тЖТ    80тЖТ  const h = sourceCanvas.height;
    81тЖТ    81тЖТ  const out = document.createElement('canvas');
    82тЖТ    82тЖТ  out.width = w;
    83тЖТ    83тЖТ  out.height = h;
    84тЖТ    84тЖТ  const ctx = out.getContext('2d')!;
    85тЖТ    85тЖТ
    86тЖТ    86тЖТ  // Draw original
    87тЖТ    87тЖТ  ctx.drawImage(sourceCanvas, 0, 0);
    88тЖТ    88тЖТ
    89тЖТ    89тЖТ  // Get pixel data
    90тЖТ    90тЖТ  const imageData = ctx.getImageData(0, 0, w, h);
    91тЖТ    91тЖТ  const data = imageData.data;
    92тЖТ    92тЖТ
    93тЖТ    93тЖТ  // Greyscale + contrast boost
    94тЖТ    94тЖТ  const contrastFactor = 1.5; // 1.0 = no change
    95тЖТ    95тЖТ  const intercept = 128 * (1 - contrastFactor);
    96тЖТ    96тЖТ  for (let i = 0; i < data.length; i += 4) {
    97тЖТ    97тЖТ    const grey = 0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2];
    98тЖТ    98тЖТ    const val = Math.min(255, Math.max(0, grey * contrastFactor + intercept));
    99тЖТ    99тЖТ    // Simple threshold for cleaner text
   100тЖТ   100тЖТ    const final = val > 140 ? 255 : 0;
   101тЖТ   101тЖТ    data[i] = data[i + 1] = data[i + 2] = final;
   102тЖТ   102тЖТ  }
   103тЖТ   103тЖТ  ctx.putImageData(imageData, 0, 0);
   104тЖТ   104тЖТ
   105тЖТ   105тЖТ  return out;
   106тЖТ   106тЖТ}
